47 void AUE4StateMachinePlayerController::MoveToMouseCursor()
48 {
49  if (UHeadMountedDisplayFunctionLibrary::IsHeadMountedDisplayEnabled())
50  {
51  if (AUE4StateMachineCharacter* MyPawn = Cast<AUE4StateMachineCharacter>(GetPawn()))
52  {
53  if (MyPawn->GetCursorToWorld())
54  {
55  UAIBlueprintHelperLibrary::SimpleMoveToLocation(this, MyPawn->GetCursorToWorld()->GetComponentLocation());
56  }
57  }
58  }
59  else
60  {
61  // Trace to see what is under the mouse cursor
62  FHitResult Hit;
63  GetHitResultUnderCursor(ECC_Visibility, false, Hit);
64 
65  if (Hit.bBlockingHit)
66  {
67  // We hit something, move there
68  SetNewMoveDestination(Hit.ImpactPoint);
69  }
70  }
71 }
